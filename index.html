ProfitChain System Plan

Income Distribution (100%)

  • 30% → Direct Referral Income: Paid instantly to the direct referrer.
  • 30% → Upline Income: 1% paid to each of the 15 uplines. If an upline doesn't exist, the share goes to the owner.
  • 30% → Downline Income: Paid to the direct referrer to support their team.
  • 10% → Reserved Pools & Fees:
    • 5% to the Reward Pool.
    • 3% to the Reality Income Pool.
    • 2% to the System Owner Wallet.

🎁 Reward System

Unlocks at Level 8. You receive a share from the 5% Reward Pool until you've earned 2x your total investment (from Levels 1-8).

🏆 Reality Income

Unlocks at Level 10. You receive a share from the 3% Reality Income Pool until you've earned 3x your total investment (from Levels 1-10).

Smart Contracts (Deploy using Remix IDE)

contract POLToken is ERC20, Ownable {
    constructor() ERC20("ProfitChain Token", "POL") {
        _mint(msg.sender, 1000000000 * 10**decimals());
    }

    function mint(address to, uint256 amount) public onlyOwner {
        _mint(to, amount);
    }
}

2. ProfitChain.sol


// SPDX-License-Identifier: MIT
pragma solidity ^0.8.9;

// You need to import this in Remix IDE:
// import "@openzeppelin/contracts/token/ERC20/IERC20.sol";

contract ProfitChain {
    address public owner;
    IERC20 public polToken;

    struct User {
        uint256 id;
        address userAddress;
        address referrer;
        uint256 activationTimestamp;
        uint256 currentLevel;
        uint256 totalEarnings;
        uint256 totalInvestment;
        uint256 directReferralsCount;
    }

    mapping(address => bool) public isUserActive;
    mapping(address => User) public users;
    mapping(uint256 => address) public idToAddress;
    mapping(address => uint256) public addressToId;

    uint256 public nextUserId = 1;
    uint256 public totalCommunitySize;

    uint256[] public levelPrices = [
        100 ether, 200 ether, 300 ether, 400 ether, 500 ether,
        600 ether, 700 ether, 800 ether, 900 ether, 1000 ether
    ];

    address public constant SYSTEM_OWNER_WALLET = 0xD5B0780E196A7D0ac60BfA057576B5c91d8F3826;
    address public rewardPoolAddress;
    address public realityPoolAddress;

    event UserRegistered(uint256 indexed id, address indexed user, address indexed referrer);
    event LevelUpgraded(address indexed user, uint256 newLevel, uint256 price);
    
    // Simple contract to hold pool funds
    contract Pool { constructor() {} }

    constructor(address _polTokenAddress) {
        owner = msg.sender;
        polToken = IERC20(_polTokenAddress);
        rewardPoolAddress = address(new Pool());
        realityPoolAddress = address(new Pool());
    }

    function register(address _referrer) public {
        require(!isUserActive[msg.sender], "User already registered");
        if (totalCommunitySize == 0) {
            require(_referrer == owner, "First user must be referred by owner");
        } else {
            require(isUserActive[_referrer], "Referrer not active");
        }

        uint256 registrationFee = levelPrices[0];
        require(polToken.transferFrom(msg.sender, address(this), registrationFee), "Token transfer failed");

        users[msg.sender] = User(nextUserId, msg.sender, _referrer, block.timestamp, 1, 0, registrationFee, 0);
        isUserActive[msg.sender] = true;
        idToAddress[nextUserId] = msg.sender;
        addressToId[msg.sender] = nextUserId;
        nextUserId++;
        totalCommunitySize++;
        if(isUserActive[_referrer]) { users[_referrer].directReferralsCount++; }
        
        distributeIncome(registrationFee, _referrer);
        emit UserRegistered(users[msg.sender].id, msg.sender, _referrer);
    }
    
    function upgradeLevel() public {
        require(isUserActive[msg.sender], "Not an active user");
        User storage user = users[msg.sender];
        require(user.currentLevel < 10, "Max level");
        uint256 nextLevel = user.currentLevel + 1;
        uint256 upgradeFee = levelPrices[nextLevel - 1];
        require(polToken.transferFrom(msg.sender, address(this), upgradeFee), "Token transfer failed");
        user.currentLevel = nextLevel;
        user.totalInvestment += upgradeFee;
        distributeIncome(upgradeFee, user.referrer);
        emit LevelUpgraded(msg.sender, nextLevel, upgradeFee);
    }

    function distributeIncome(uint256 _amount, address _referrer) private {
        uint256 directIncome = (_amount * 30) / 100;
        sendIncome(_referrer, directIncome);
        users[_referrer].totalEarnings += directIncome;
        
        uint256 uplineChunk = (_amount * 1) / 100;
        address currentUpline = _referrer;
        for (uint i = 0; i < 15; i++) {
            if (currentUpline != address(0) && isUserActive[currentUpline]) {
                sendIncome(currentUpline, uplineChunk);
                users[currentUpline].totalEarnings += uplineChunk;
                currentUpline = users[currentUpline].referrer; 
            } else { sendIncome(owner, uplineChunk); }
        }
        
        uint256 downlineIncome = (_amount * 30) / 100;
        sendIncome(_referrer, downlineIncome);
        users[_referrer].totalEarnings += downlineIncome;

        polToken.transfer(rewardPoolAddress, (_amount * 5) / 100);
        polToken.transfer(realityPoolAddress, (_amount * 3) / 100);
        polToken.transfer(SYSTEM_OWNER_WALLET, (_amount * 2) / 100);
    }
    
    function sendIncome(address _to, uint256 _amount) private {
        if (_amount > 0) { polToken.transfer(_to, _amount); }
    }

    function getUserDetails(address _userAddress) public view returns (User memory, string memory, uint256, uint256) {
        require(isUserActive[_userAddress], "User not found");
        User memory user = users[_userAddress];
        string memory referralLink = string(abi.encodePacked("YOUR_WEBSITE_URL/?ref=", _uintToAsciiString(user.id)));
        return (user, referralLink, polToken.balanceOf(rewardPoolAddress), polToken.balanceOf(realityPoolAddress));
    }
    
    function _uintToAsciiString(uint256 _number) private pure returns (string memory) {
        if (_number == 0) return "0";
        uint256 temp = _number; uint256 digits;
        while (temp != 0) { digits++; temp /= 10; }
        bytes memory buffer = new bytes(digits);
        while (_number != 0) { digits -= 1; buffer[digits] = bytes1(uint8(48 + _number % 10)); _number /= 10; }
        return string(buffer);
    }
}
